## useEffect

**useEffect用来替代生命周期函数**

```tsx
// useEffect在组件渲染和状态更新时调用，相当于componentDivMount和componentUpdateMount
import React from 'react'

export default function Home() {
  const [count,setCount] = React.useState(0);
  React.useEffect(()=>{
    console.log("这是首页");
  })
  return (
    <div>
      这是Home页面
      <p>Cliced is {count}</p>
      <button onClick={()=>{setCount(count+1)}}>Clicked me</button>
    </div>
  )
}
// 若要实现类似 componentWillUnmount的效果，则需要加入第二个参数和return
React.useEffect(()=>{
    console.log("这是首页");
    return () => {console.log("离开首页");}
},[])
```
