## useContext

**useContext用于父子组件的传值**

```tsx
import React, { useState,useContext,createContext } from 'react';

//创建createContext
const CountContext = createContext({});

//子组件
function ChilderExample() {
  //父组件使用useContext接收
  const info:any = useContext(CountContext);
  return (
    <h1>{info[0]}</h1>
  )
}

//父组件
function Example4() {
  const [count,setCount] = useState(0);
  const [index,setIndex] = useState(0);
  return (
    <div>
      <p> Clicked is {count} timer</p>
      <button onClick={()=>{setCount(count+1)}}>Click me</button>
      {/*在要引入的地方使用CountContext.Provider引入*/}
      <CountContext.Provider value={[count,index]}>
        <ChilderExample />
      </CountContext.Provider>
    </div>
  )
}

export default Example4

```
