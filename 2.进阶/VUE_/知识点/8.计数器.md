    <div id="app">
        <div class="input-num">
            <button @click="sub">-</button>
            <span>{{num}}</span>
            <button @click="add">+</button>
        </div>
    </div>
    <script>
        let app = new Vue({
            el:"#app",
            data:{
                num: 0
            },
            methods:{
                add: function(){
                    if(this.num >= 10){
                        alert("到头啦!");
                        return;
                    }
                    this.num++;
                },
                sub: function(){
                    if(this.num <= 0){
                        alert("不能再少啦!");
                        return;
                    }
                    this.num--;
                }
            }
        })
    </script>
</body>
