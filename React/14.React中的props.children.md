## 1.props.children

**组件标签也能像普通的 HTML 标签那样编写内嵌的结构，并通过props.children获取到**

```react
import React from 'react';
import ReactDOM from 'react-dom';

class Card extends React.Component {
    render () {
        console.log(this.props.children);
      return (
        <div className='card'>
          <div className='card-content'>
            {this.props.children[0]}
          </div>
          <div className='card-content'>
            {this.props.children[1]}
          </div>
        </div>
      )
    }
  }
  ReactDOM.render(
    <Card>
        <div>
            <h2>React.js 小书</h2>
            <div>开源、免费、专业、简单</div>
            订阅：<input />
        </div>
        <div>
            <h2>React.js 小书2</h2>
            <div>开源、免费、专业、简单</div>
            订阅：<input />
        </div>
    </Card>,
    document.getElementById('root')
  )
```
