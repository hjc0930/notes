#### 1.class语法：
```javascript
//传统构造函数模仿类的写法
function Person(name,age,sex){
    this.name = name;
    this.sex = sex;
    this.age = age;
}
Person.prototype.showSelf = function(){
    alert(`我叫${this.name},今年${this.age}岁,是一个${this.sex}孩`);
}

//ECMA6类的写法
class Person{
    constructor(name,age,sex){
        this.name = name;
        this.age = age;
        this.sex = sex;
    }
    showSelf(){
        alert(`我叫${this.name},今年${this.age}岁,是一个${this.sex}孩`);
    }
}
//调用
let p1 = new Person("blue",18,"男");
p1.showSelf();
```
